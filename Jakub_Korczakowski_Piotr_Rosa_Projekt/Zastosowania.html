function make_moon(; n_samples = 100, shuffle = true)
    n_samples_out = n_samples  2
    n_samples_in = n_samples - n_samples_out

    X = zeros(2,n_samples)
    y = zeros(n_samples)

    # outer circle
    X[1,1:n_samples_out] = cos.(LinRange(0, , n_samples_out))
    X[2,1:n_samples_out] = sin.(LinRange(0, , n_samples_out))
    y[1:n_samples_out] .= 0

    # inner circle
    X[1,n_samples_out+1:end] = 1 .- cos.(LinRange(0, , n_samples_in))
    X[2,n_samples_out+1:end] = 1 .- sin.(LinRange(0, , n_samples_in)) .- 0.5
    y[n_samples_out+1:end] .= 1

    if shuffle
       ind = randperm(n_samples)
       return X[:,ind],y[ind]
    else
       return X,y
    end
end

function my_svd(A::Matrix; maxiter=1000)
    V = randn(size(A,2), size(A,1)) # random initialization
    for _ = 1:maxiter
        W = A * V
        Z = A' * W
        V, R = mgramschmidt(Z)
    end
    W = A * V
    S = [norm(W[:, i]) for i=1:size(W,2)] # singular values
    U = reduce(hcat,[W[:,i]/S[i] for i=1:size(W,2)]) # left singular vectors
    return U, S, V
end

function my_pca(A::Matrix; maxiter=1000)
    C = A .- mean(A);
    , V = jacobi_B_vect(cov(C'), maxiter);
    W = V;
    P = W * A;
end

function kernel_PCA(X,kernel,ncomp)
    N = size(X,2)
    K = zeros(N,N)

   for j = 1:N
      for i = 1:N
        K[i,j] = kernel(X[:,i],X[:,j])
      end
    end
    one_n = ones(N,N)/N
    # recenter K
    Kp = Symmetric(K - one_n * K - K * one_n + one_n * K * one_n)
    Kp = convert(Matrix, Kp)
    
    , V = jacobi_B_vect(Kp, 10000);

    V = V[:, end-ncomp+1:end];
    V = reverse(V, dims=2)
     = [end-ncomp+1:end]
     = reverse()

    [ .< 0] .= 0
     = /N
    amplitudes = Diagonal(sqrt.()) * V';

    return amplitudes, , V, K
end
